通过 AI 实现自动生成 SEO TDK

06 Mar, 2025

感谢博友的文章，《AI 实践｜零成本生成 SEO 友好的 TDK 落地方案》。

昨天晚上刷到一篇文章，通过 AI 自动化生成网站 SEO 的 TDK，他这个方法有两点比较繁琐的地方。

  1. 文章内容需要手动替换到脚本中
  2. 生成的 TDK 需要手动加入数据库

我针对第一点略微做了一些调整，可以直接查找当前脚本目录下的文件夹，在文件的 62 行，稍后我会把源码贴在文章最下方。

https://github.com/dlzmoe/blog/blob/main/aitdk.py

1# aitdk.py
2def main(file_name, api_key, api_url, model, debug=False):
3    current_dir = os.path.dirname(os.path.abspath(__file__))
4    file_path = os.path.join(current_dir, 'content', 'blog', file_name)

这里的目录 /content/blog 可以自由修改，换成你本地的格式。

使用方法也很简单，在 aitdk.py 同级目录下新建一个 .env 环境变量文件，内容如下：

1OPENAI_API_KEY=sk-xxx
2OPENAI_API_URL=https://api.openai.com/v1
3OPENAI_API_MODEL=gpt-4o-mini

然后安装一下依赖。

1pip install markdown beautifulsoup4 openai==0.28 python-dotenv

运行命令，xxx.md 就是你的文件全名。

1py aitdk.py xxx.md

生成的结果就如下，对于 SEO 质量还是不错的。

 1import os
 2import json
 3import markdown
 4from bs4 import BeautifulSoup
 5import openai
 6from dotenv import load_dotenv
 7import argparse
 8
 9# 加载环境变量
10load_dotenv()
11
12def extract_text_from_markdown(file_path):
13    with open(file_path, 'r', encoding='utf-8') as file:
14        md_content = file.read()
15    
16    html_content = markdown.markdown(md_content)
17    soup = BeautifulSoup(html_content, 'html.parser')
18    text_content = soup.get_text()
19    
20    return text_content
21
22def generate_seo_content(text, api_key, api_url, model, debug=False):
23    openai.api_key = api_key
24    if api_url:
25        openai.api_base = api_url
26
27    prompt = f"请根据文章内容从 SEO 友好的角度提取出标题、关键词和描述:\n\n{text}\n\n请以 JSON 格式输出，包含 slug、title、keywords 和 description 字段。"
28    
29    try:
30        response = openai.ChatCompletion.create(
31            model=model,
32            messages=[
33                {"role": "system", "content": "你是一个 SEO 专家，擅长提炼文章的核心内容并生成优化的元数据。"},
34                {"role": "user", "content": prompt}
35            ]
36        )
37        
38        content = response.choices[0].message['content']
39        # 只在 debug 模式下输出 API 响应
40        if debug:
41            print("API Response:", content)
42        return content
43    except Exception as e:
44        print(f"Error calling OpenAI API: {e}")
45        return None
46
47def parse_seo_content(content):
48    try:
49        # 尝试直接解析 JSON
50        return json.loads(content)
51    except json.JSONDecodeError:
52        # 如果直接解析失败，尝试提取 JSON 部分
53        try:
54            start = content.index('{')
55            end = content.rindex('}') + 1
56            json_str = content[start:end]
57            return json.loads(json_str)
58        except (ValueError, json.JSONDecodeError):
59            print("无法解析 API 返回的内容为 JSON 格式")
60            return None
61
62def main(file_name, api_key, api_url, model, debug=False):
63    current_dir = os.path.dirname(os.path.abspath(__file__))
64    file_path = os.path.join(current_dir, 'content', 'blog', file_name)
65    
66    if not os.path.exists(file_path):
67        print(f"错误：文件 '{file_path}' 不存在。")
68        return
69
70    text_content = extract_text_from_markdown(file_path)
71    seo_content = generate_seo_content(text_content, api_key, api_url, model, debug)
72    
73    if seo_content:
74        seo_data = parse_seo_content(seo_content)
75        if seo_data:
76            print(json.dumps(seo_data, ensure_ascii=False, indent=2))
77        else:
78            print("无法生成有效的 SEO 数据")
79    else:
80        print("生成 SEO 内容失败")
81
82if __name__ == "__main__":
83    parser = argparse.ArgumentParser(description="从 Markdown 文件生成 SEO 内容")
84    parser.add_argument("file_name", help="Markdown 文件名称 (位于 content/blog/目录下)")
85    parser.add_argument("--api_key", default=os.getenv("OPENAI_API_KEY"), help="OpenAI API 密钥")
86    parser.add_argument("--api_url", default=os.getenv("OPENAI_API_URL"), help="OpenAI API URL")
87    parser.add_argument("--model", default=os.getenv("OPENAI_API_MODEL"), help="OpenAI 模型名称")
88    parser.add_argument("--debug", action="store_true", help="开启调试模式，显示 API 响应")
89    
90    args = parser.parse_args()
91
92    if not args.api_key:
93        print("错误：未提供 API 密钥。请在命令行参数中指定或在.env 文件中设置 OPENAI_API_KEY。")
94    else:
95        main(args.file_name, args.api_key, args.api_url, args.model, args.debug)
