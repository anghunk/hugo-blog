 3smalltoBIG (n) {
 4  var fraction = ['角', '分'];
 5  var digit = ['零', '壹', '贰', '叁', '肆', '伍', '陆', '柒', '捌', '玖'];
 6  var unit = [
 7    ['元', '万', '亿'],['', '拾', '佰', '仟']
 8  ];
 9  var head = n < 0 ? '欠' : '';
10  n = Math.abs(n);
11
12  var s = '';
13
14  for (var i = 0; i < fraction.length; i++) {
15    s += (digit[Math.floor(n * 10 * Math.pow(10, i)) % 10] + fraction[i]).replace(/零./, '');
16  }
17  s = s || '整';
18  n = Math.floor(n);
19
20  for (var i = 0; i < unit[0].length && n > 0; i++) {
21    var p = '';
22    for (var j = 0; j < unit[1].length && n > 0; j++) {
23      p = digit[n % 10] + unit[1][j] + p;
24      n = Math.floor(n / 10);
25    }
26    s = p.replace(/(零.)*零$/, '').replace(/^$/, '零') + unit[0][i] + s;
27  }
28  return head + s.replace(/(零.)*零元/, '元').replace(/(零.)+/g, '零').replace(/^整$/, '零元整');
29},
