在 vue 中制作 canvas 波浪图

29 Mar, 2022

新建组件 Canvas.vue 。

  1// @/components/Canvas.vue
  2
  3<template>
  4  <div :style="{ height: waveAllHeight + 'px', background: bgColor }" class="wave">
  5    <canvas id="wave1"></canvas>
  6    <canvas id="wave2"></canvas>
  7    <canvas id="wave3"></canvas>
  8  </div>
  9</template>
 10
 11<script>
 12export default {
 13  name: "Canvas",
 14  props: {
 15    bgColor: {
 16      default: "none",
 17    }, // 背景色
 18    waveAllHeight: { default: 160 }, // 波浪的整体高度
 19    waveCount: { default: 2 }, // 波峰个数
 20    waveHeight: { default: 50 }, // 波浪起伏高度
 21    // 波浪颜色
 22    waveColor: {
 23      default () {
 24        return [
 25          "#f3f3f3",
 26          "#f3f3f3",
 27          "#f3f3f3",
 28        ];
 29      },
 30    },
 31    // 波浪速率
 32    waveTime: {
 33      default () {
 34        return [4000, 4000, 4000];
 35      },
 36    },
 37  },
 38  data () {
 39    return {};
 40  },
 41  mounted () {
 42    this.wavePlay("wave1", 140, this.waveColor[0], this.waveTime[0]);
 43    this.wavePlay("wave2", 140, this.waveColor[1], this.waveTime[1]);
 44    this.wavePlay("wave3", 140, this.waveColor[2], this.waveTime[2]);
 45  },
 46  methods: {
 47    wavePlay ($canvasID, $progress, $maveColor, $time) {
 48      const that = this;
 49      let waveWidth = 3300, // 波浪长度
 50        offset = 0,
 51        waveHeight = that.waveHeight, // 波浪起伏高度
 52        waveCount = that.waveCount, // 波浪个数
 53        startX = -1200,
 54        startY = 212, // canvas 高度
 55        progress = $progress, // 波浪位置高度
 56        d2 = waveWidth / waveCount, // 单个波浪的宽度
 57        d = d2 / 2,
 58        hd = d / 2,
 59        c = document.getElementById($canvasID),
 60        ctx = c.getContext("2d");
 61      c.width = 1920; // 画布宽度
 62      c.height = that.waveAllHeight; // 画布高度
 63      function move () {
 64        offset -= 5;
 65        if (-1 * offset === d2) {
 66          offset = 0;
 67        }
 68        ctx.clearRect(0, 0, c.width, c.height);
 69        ctx.fillStyle = $maveColor; // 画布填充色
 70        ctx.beginPath();
 71        let offsetY = startY - progress;
 72        // 绘制贝塞尔曲线
 73        ctx.moveTo(startX - offset, offsetY); // 开始点
 74        for (let i = 0; i < waveCount; i++) {
 75          let dx = i * d2;
 76          let offsetX = dx + startX - offset;
 77          ctx.quadraticCurveTo(
 78            offsetX + hd,
 79            offsetY + waveHeight,
 80            offsetX + d,
 81            offsetY
 82          );
 83          ctx.quadraticCurveTo(
 84            offsetX + hd + d,
 85            offsetY - waveHeight,
 86            offsetX + d2,
 87            offsetY
 88          );
 89        }
 90        ctx.lineTo(startX + waveWidth, 3000);
 91        ctx.lineTo(startX, 0);
 92        ctx.fill();
 93        setTimeout(move, $time / 60); // 速度
 94      }
 95
 96      move();
 97    },
 98  },
 99};
100</script>
