利用 memos 生成的在线动态列表

09 Jan, 2023

memos 是一个具有知识管理和社交功能的开源自托管备忘录中心。

Github 地址：https://github.com/usememos/memos

可以用它本身提供的 api，然后结合静态博客，做成一个在线的动态列表，类似于朋友圈功能。

接口 url 有固定的格式，openId 是每个用户自动生成的。

其次我又写了两个方法，一个是时间戳转成正常时间格式，一个是利用正则删去内容中多余的标签。

将下面代码复制到你想显示的页面编辑，只需修改 openId，提供了一个基础的样式，如果有其他优化方案，欢迎指出。

由于接口中带了用户密钥，建议完成代码后将 js 加密再引入，不然可能会有一定的安全风险。

推荐网址：https://tool.lu/js/

 1// 调用接口的 js 方法
 2function memosShow() {
 3  var memosStr = "";
 4  var openId = "xxx";
 5  $.ajax({
 6    url: "https://memos.zburu.com/api/memo?openId=" + openId + "&tag=说说",
 7    type: "get",
 8    dataType: "json",
 9    success: function (data) {
10      // 生成数组
11      const sslist = data.data;
12
13      for (let i = 0; i < sslist.length; i++) {
14        const element = sslist[i];
15
16        // 把时间戳转为正常时间格式 2023-01-9 13:17:12
17        var date = new Date(element.createdTs * 1000);
18        Y = date.getFullYear() + "-";
19        M =
20          (date.getMonth() + 1 < 10
21            ? "0" + (date.getMonth() + 1)
22            : date.getMonth() + 1) + "-";
23        D = (date.getDate() < 10 ? "0" + date.getDate() : date.getDate()) + " ";
24        h = date.getHours() + ":";
25        m = date.getMinutes() + ":";
26        s = date.getSeconds();
27        const createdTsNew = Y + M + D + h + m + s;
28        sslist[i].createdTs = createdTsNew;
29
30        // 利用js正则删去内容前的标签字符
31        const contentNew = element.content.slice(4);
32        sslist[i].content = contentNew;
33      }
34
35      $.each(data.data, function (i, item) {
36        list =
37          "<div class='sslist-item'>" +
38          "<p class='sslist-date'>" +
39          item.createdTs +
40          "</p>" +
41          item.content +
42          "</div>";
43        memosStr += list;
44      }),
45        $(".sslist").html(memosStr);
46    },
47    error: function () {
48      console.log("error");
49    }
50  });
51}
52memosShow();
