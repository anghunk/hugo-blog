笔记:vue 中使用 axios 调用数据并渲染

12 Jul, 2022

示例如下。

 1<template>
 2  <div class="posts">
 3    <div
 4      v-for="item in list"
 5      :key="item.cid"
 6    >
 7      {{ item.title }}
 8    </div>
 9  </div>
10</template>
11
12<script>
13import Axios from 'axios';
14export default {
15  name: 'index',
16  data () {
17    return {
18      list: []
19    }
20  },
21  methods: {
22    getData () {
23      var api = 'https://zburu.com/api/posts.php';
24      Axios.get(api).then((response) => {
25        this.list = response.data;
26        console.log(response.data)
27      }).catch((error) => {
28        console.log(error);
29      })
30    }
31  },
32  mounted () {
33    this.getData();
34  },
35}
36</script>
