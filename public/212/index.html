python 通过 json 生成小文件

10 Nov, 2023

记录一个方法，pyhton 通过 json 文件，在同级目录下生成对应格式的小文本。

 1import json
 2
 3def generate_files_from_json(json_file):
 4    with open(json_file, 'r', encoding='utf-8') as f:
 5        data = json.load(f)
 6
 7    for item in data:
 8        for key, value in item.items():
 9            if isinstance(value, list):
10                for sub_item in value:
11                    for sub_key, sub_value in sub_item.items():
12                        if isinstance(sub_value, dict) and 'list' in sub_value:
13                            for sub_sub_item in sub_value['list']:
14                                if 'id' in sub_sub_item and 'css' in sub_sub_item:
15                                    id_value = sub_sub_item['id']
16                                    css_value = sub_sub_item['css']
17                                    create_file(id_value, css_value)
18
19def create_file(file_name, content):
20    file_path = f"{file_name}.css"
21    with open(file_path, 'w', encoding='utf-8') as file:
22        file.write(content)
23
24    print(f"File '{file_path}' created successfully.")
25
26if __name__ == "__main__":
27    json_file_path = 'your_json_file.json'
28    generate_files_from_json(json_file_path)
