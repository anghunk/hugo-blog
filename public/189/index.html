用原生 js 给网站写个搜索功能

17 Mar, 2023

本文也算是一篇教程，可以给 hugo 网站加个搜索功能，并且实现热更新，体验感更好。

如果是其他程序，只需要按照特定的模板生成以下格式的文件即可，主要代码从 第 2 部分 开始，第一章节写的是如何用 hugo 输出文章列表 json 文件。

title 是文章标题，permalink 是文章链接。

1[{
2  "permalink": "",
3  "title": ""
4}, {
5  "permalink": "",
6  "title": ""
7}]

1. hugo 模板生成文章列表 json 文件

在 layouts 文件夹下新建 index.json 文件，模板内容如下：

其中第 2 行最后面的 "blog" 是你文章文件夹的名称。大部分是 posts 等等，这里是我个人的名称。

1{{- $.Scratch.Set "posts" slice -}}
2{{- range where .Site.RegularPages "Type" "blog" -}}
3    {{- $.Scratch.Add "posts" (dict "title" .Title "permalink" .Permalink) -}}
4{{- end -}}
5{{- $.Scratch.Get "posts" | jsonify -}}

按照这个模板，hugo 本地预览可以打开 http://localhost:1313/index.json 查看，如果输出了一些数据如下图，说明你成功了。

2. js 代码

在 /layouts/_default 新建一个模板文件 search.html，大致的结构参考其他模板文件，然后写入我们需要的内容。

首先是一个简单的 html 结构，给 input 绑定一个事件。

1<form class="search"> 
2  <input type="text" id="searchTerm" name="searchTerm" autocomplete="off" oninput="initiateSearch()">
3</form>
4<div id="resultsContainer">请输入关键词进行搜索...</div>

然后通过一个 get 请求获取 json 文件，传入关键词参数，生成搜索列表。

 1<script>
 2  function search(jsonData, searchTerm) {
 3    let results = [];
 4    for (let i = 0; i < jsonData.length; i++) {
 5      for (let property in jsonData[i]) {
 6        if (jsonData[i].hasOwnProperty(property) && jsonData[i][property].toString().indexOf(searchTerm) > -1) {
 7          results.push(jsonData[i]);
 8          break;
 9        }
10      }
11    }
12    return results;
13  }
14
15  function displayResults(searchResults) {
16    let container = document.getElementById("resultsContainer");
17    container.innerHTML = "";
18    if (searchResults.length > 0) {
19      for (let i = 0; i < searchResults.length; i++) {
20        let resultDiv = document.createElement("div");
21        let resultTitle = document.createElement("a");
22        resultTitle.innerText = searchResults[i].title;
23        resultTitle.setAttribute('href', searchResults[i].permalink)
24        resultDiv.appendChild(resultTitle);
25        container.appendChild(resultDiv);
26      }
27    } else {
28      let noResultsMessage = document.createElement("p");
29      noResultsMessage.innerText = "没有找到搜索结果。";
30      container.appendChild(noResultsMessage);
31    }
32  }
33
34  function initiateSearch() {
35    let searchTerm = document.getElementById("searchTerm").value;
36    let xhr = new XMLHttpRequest();
37    xhr.open('GET', '/index.json', true);
38    xhr.onreadystatechange = () => {
39      if (xhr.readyState === 4 && xhr.status === 200) {
40        let jsonData = JSON.parse(xhr.responseText);
41        let searchResults = search(jsonData, searchTerm);
42        displayResults(searchResults);
43      }
44    };
45    xhr.send();
46  }
47</script>

然后再在 /content 新建一个 search.md 文件调用该模板即可。

1---
2slug: search
3title: 搜索
4layout: search
5---
