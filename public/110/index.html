增加了站内搜索功能

25 Mar, 2022

如题，我花了一个多小时，参照 nuxt 文档，把搜索功能给加上了，记录一下相关代码。

可以选择放在 components/<Search /> 里面，然后引入到 head 模板，也可以单独创建一个页面，我选择的是后者。

新建一个页面模板 /pages/search.vue 。

 1<template>
 2  <div>
 3    <HeaderMe />
 4    <div class="wrapper">
 5      <PageSidebar />
 6      <div class="archive">
 7        <h2>{{ article.attributes.title }}</h2>
 8        <div class="article-content markdown-body" v-html="article.html"></div>
 9        <div class="search">
10          <input v-model="searchQuery" type="search" autocomplete="off" placeholder="文章关键词" />
11          <ul v-if="articles.length">
12            <li v-for="article of articles" :key="article.slug">
13              <NuxtLink :to="article.path+'/'">{{ article.title }}</NuxtLink>
14               <!-- 在url地址后加一个/，配置全站的url格式，防止错乱 -->
15            </li>
16          </ul>
17        </div>
18      </div>
19    </div>
20  </div>
21</template>
22
23<script>
24import HeaderMe from "@/components/HeaderMe";
25import PageSidebar from '@/components/PageSidebar';
26export default {
27  components: {
28    HeaderMe,
29    PageSidebar
30  },
31  data () {
32    return {
33      searchQuery: '',
34      articles: []
35    }
36  },
37  async asyncData () {
38    // 调用 search.md 内的数据
39    const article = await import(`~/content/search.md`);
40    return {
41      article
42    };
43  },
44  watch: {
45    async searchQuery (searchQuery) {
46      if (!searchQuery) {
47        this.articles = []
48        return
49      }
50      this.articles = await this.$content('posts') // posts 是文章所在的文件夹的名称
51        .limit(10)
52        .search(searchQuery)
53        .fetch()
54    }
55  }
56}
57</script>
