使用 jq 做一个数字递增效果

14 Sep, 2021

数字递增，顾名思义就是数字不断增加，数字递增的效果就是要一定时间内，让数字有一个增加的特效，一般用于强调某个内容。

用 jquery 也很好处理，大概就是下面这些。

  • data-to 属性代表最后想要递增到的数值
  • data-speed 代表数值想要递增的过程时间

class="num-ber" 和 id="count-number" 按照这个填入。

1<p>
2  <span class="num-ber" id="count-number" data-to="40" data-speed="1000"></span>+
3</p>

引入 jquery 之后再添加下面的 js 代码。

 1$.fn.countTo = function (a) {
 2  a = a || {};
 3  return $(this).each(function () {
 4    var c = $.extend({},
 5      $.fn.countTo.defaults, {
 6        from: $(this).data("from"),
 7        to: $(this).data("to"),
 8        speed: $(this).data("speed"),
 9        refreshInterval: $(this).data("refresh-interval"),
10        decimals: $(this).data("decimals")
11      }, a);
12    var h = Math.ceil(c.speed / c.refreshInterval),
13      i = (c.to - c.from) / h;
14    var j = this,
15      f = $(this),
16      e = 0,
17      g = c.from,
18      d = f.data("countTo") || {};
19    f.data("countTo", d);
20    if (d.interval) {
21      clearInterval(d.interval)
22    }
23    d.interval = setInterval(k, c.refreshInterval);
24    b(g);
25
26    function k() {
27      g += i;
28      e++;
29      b(g);
30      if (typeof (c.onUpdate) == "function") {
31        c.onUpdate.call(j, g)
32      }
33      if (e >= h) {
34        f.removeData("countTo");
35        clearInterval(d.interval);
36        g = c.to;
37        if (typeof (c.onComplete) == "function") {
38          c.onComplete.call(j, g)
39        }
40      }
41    }
42
43    function b(m) {
44      var l = c.formatter.call(j, m, c);
45      f.html(l)
46    }
47  })
48};
49$.fn.countTo.defaults = {
50  from: 0,
51  to: 0,
52  speed: 1000,
53  refreshInterval: 100,
54  decimals: 0,
55  formatter: formatter,
56  onUpdate: "null,
57  onComplete: null
58};
59
60function formatter(b, a) {
61  return b.toFixed(0)
62}
63$("#count-number").data("countToOptions", {
64  formatter: function (b, a) {
65    return b.toFixed(0).replace(/\B(?=(?:\d{3})+(?!\d))/g, ",")
66  }
67});
68$(".num-ber").each(count);
69
70function count(a) {
71  var b = $(this);
72  a = $.extend({},
73    a || {},
74    b.data("countToOptions") || {});
75  b.countTo(a)
76};
