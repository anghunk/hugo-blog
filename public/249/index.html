使用 wxt 框架开发浏览器扩展时遇到的问题解决方案

25 Dec, 2024

最近在使用 wxt 框架开发浏览器扩展，记录一些问题。

https://wxt.dev/

1. 隔离脚本与网站本身的样式

使用 Shadow Root，可以不让 content-scripts 侵入式影响网站本身，然后引入并渲染 App.vue 组件。

 1import './style.css';
 2import { createApp } from 'vue';
 3import App from './App.vue';
 4
 5export default defineContentScript({
 6  matches: ['https://google.com/*'],
 7  cssInjectionMode: 'ui',
 8
 9  async main(ctx) {
10    const ui = await createShadowRootUi(ctx, {
11      name: 'example-ui',
12      position: 'inline',
13      anchor: 'body',
14      onMount: (container) => {
15        const app = createApp(App);
16        app.mount(container);
17        return app;
18      },
19      onRemove: (app) => {
20        app?.unmount();
21      },
22    });
23
24    ui.mount();
25  },
26});

2. 在 config.js 中配置

 1import { defineConfig } from 'wxt';
 2import pkg from './package.json';
 3
 4// See https://wxt.dev/api/config.html
 5export default defineConfig({
 6  extensionApi: 'chrome',
 7  modules: ['@wxt-dev/module-vue'],
 8  manifest: {
 9    name: 'xxx',
10    version: pkg.version,
11    description: ...,
12    permissions: ['storage', 'scripting'] // 存储、插入脚本
13  },
14  hooks: {
15    'build:manifestGenerated': (wxt, manifest) => {
16      if (wxt.config.command === 'serve') {
17        manifest.content_scripts ??= [];
18        manifest.content_scripts.push({
19          matches: ['https://google.com/*'],
20          js: ['content-scripts/content.js'],
21          css: ['content-scripts/content.css']
22        });
23      }
24    }
25  }
26});
